-- Tabela "Cliente"
CREATE TABLE IF NOT EXISTS Cliente( 
                     ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                     Nome VARCHAR(100),
                     Email VARCHAR(100),
                     Telefone  VARCHAR(18),
         	     AvatarUrl TEXT					 
                     );   

-- Tabela "Fornecedor" 
CREATE TABLE IF NOT EXISTS Fornecedor( 
                     ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                     Nome VARCHAR(100),
                     Email VARCHAR(100),
                     Telefone VARCHAR(18),
					 AvatarUrl TEXT
                     );   

-- Tabela "Categoria" 
CREATE TABLE IF NOT EXISTS Categoria( 
                     ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                     Categoria_Descricao VARCHAR(100));   
				
-- Tabela "Receitas"
CREATE TABLE IF NOT EXISTS receitas( 
                     ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                     ID_Cliente INTEGER,
                     NParcela INT,
                     Valor FLOAT (15,2),
                     DataVencimento DATETIME,
                     ID_Categoria INTEGER,
                     Observacoes TEXT);

-- Tabela "ParcelaReceita"					 
CREATE TABLE IF NOT EXISTS ParcelaReceita(ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                           ID_Cliente INTEGER,
                           ID_Receita INTEGER,
                           NParcela INT,
                           Valor FLOAT (15,2),
                           ValorPago FLOAT (15,2),
                           DataVencimento DATETIME,
                           DataPagamento DATETIME, 
                           ID_Categoria INTEGER,
                           Observacoes TEXT);     

-- Tabela "Despesas" 
CREATE TABLE IF NOT EXISTS despesas( 
                     ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                     ID_Fornecedor INTEGER,
                     NParcela INT,
                     Valor FLOAT (15,2),
                     DataVencimento DATETIME,
                     ID_Categoria INTEGER,
                     Observacoes TEXT);

-- Tabela "ParcelaDespesa"                  						  
CREATE TABLE IF NOT EXISTS ParcelaDespesa(ID INTEGER PRIMARY KEY AUTOINCREMENT,  
                           ID_Fornecedor INTEGER,
                           ID_Despesa INTEGER,
                           NParcela INT,
                           Valor FLOAT (15,2),
                           ValorPago FLOAT (15,2),
                           DataVencimento DATETIME,
                           DataPagamento DATETIME, 
                           ID_Categoria INTEGER,
                           Observacoes TEXT);
